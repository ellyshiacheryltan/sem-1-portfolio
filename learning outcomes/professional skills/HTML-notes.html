<a href="#section1">Section 1</a> <!-- access the section by its ID -->

<!-- section 1 -->
<section class="section1" id="section1">
   <p>Hello world!</p>

2 Use <label> to expand the accessibility of the <input> element

  • example of usage:
<html>
    <head>
        <link rel="stylesheet" href="style.css">
        <title>Image Slideshow</title>
    </head>
    <body>

        <!-- elements container -->
        <div class="hero">

            <!-- image slides container -->
            <div class="slides">
                
                <!-- the 'name' attribute is to group the related radio buttons together, so only 1 radio button can be
                    selected at a time -->
                <input type="radio" name="selector" id="button1">
                <input type="radio" name="selector" id="button2">
                <input type="radio" name="selector" id="button3">

                <div class="image first">
                    <img src="images/slide_1.jpg">
                </div>
    
                <div class="image">
                    <img src="images/slide_2.jpg">
                </div>
    
                <div class="image">
                    <img src="images/slide_3.jpg">
                </div>

                <!-- directly connected to the radio buttons to change the input -->
                <div class="navigator">

                    <!-- labels increase hit area of buttons -->
                    <label for="button1" class="bar"></label>
                    <label for="button2" class="bar"></label> 
                    <label for="button3" class="bar"></label>  
                </div>

            </div>

        </div>
    </body>
</html>

3 Difference between absolute and relative file paths

<!-- absolute path = full url to a file -->
<img src="https://webURL/images/picture.jpg" alt="image"/>

<!-- relative path = path to a file relative to the current page -->
<img src="images/picture.jpg" alt="image"/>

4 Displaying a highlighted markdown with highlight.js

<html>
	<head>
		<link rel="stylesheet" href="https://cdnjs.cloudflare.com/ajax/libs/highlight.js/11.9.0/styles/default.min.css">
	</head>
	<body>

		<!-- example of JavaScript markdown -->
		<pre><code class="javascript">
			function greet(name) {
			  console.log(`Hello, ${name}!`);
			}
		</code></pre>

		<br>

		<!-- example of HTML markdown -->
		<pre><code class="html">
			&lt;h1&gt;Hello, world!&lt;/h1&gt;
		</code></pre>

		<!-- highlight.js scripts -->
		<script src="https://cdnjs.cloudflare.com/ajax/libs/highlight.js/11.9.0/highlight.min.js"></script>
    <script>hljs.initHighlightingOnLoad();</script>
	</body>
</html>
